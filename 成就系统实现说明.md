## 二、解锁检测函数（TypeScript）

```typescript
// ============================================================
// 关键定义
// ============================================================
// 「打卡」    = 当日学习总时长 ≥ 30 分钟
// 「超额」    = 该科当日时长 ≥ 该科每日目标 × 1.5
//              （物理 2h→3h，数学 2h→3h，英语 2.5h→3.75h）
// 「晨行」    = 当天第一条学习记录的开始时间 < 8:00
// 「夜航」    = 当天最后一条学习记录的结束时间 > 22:00
// 「单段连续」= 合并后时间轴上相邻记录间隔 < 5min 的归为一段
// 「逆袭」    = 前一天总时长 < prevMax（分钟），今天 ≥ todayMin（分钟）
// 「均衡日」  = 三科全部 ≥ 各自基础目标，且全部 < 目标 × 1.5
// 「朝暮行」  = 同一天同时满足晨行 + 夜航

// ============================================================
// 检测入口：每次数据刷新后调用
// ============================================================
function checkAchievements(
  dailyData: DayData,          // 今天的完整数据
  allHistory: DayData[],       // 暑假以来所有天的数据
  subjectGoals: { physics: number; math: number; english: number }
): Achievement[] {
  const unlocked: Achievement[] = [];

  const allAchievements = [...ACHIEVEMENTS, ...HIDDEN_ACHIEVEMENTS];
  const alreadyUnlocked = getUnlockedIds(); // 从 DB 读已解锁的 id

  for (const ach of allAchievements) {
    if (alreadyUnlocked.has(ach.id)) continue;

    let satisfied = false;

    switch (ach.condition.metric) {
      // ── 累计类 ──
      case 'totalHours':
        satisfied = sumTotalHours(allHistory) >= ach.condition.threshold;
        break;

      case 'subjectHours':
        satisfied = sumSubjectHours(allHistory, ach.condition.subject!) >= ach.condition.threshold;
        break;

      case 'streakDays':
        satisfied = getConsecutiveDays(allHistory) >= ach.condition.threshold;
        break;

      // ── 单日类 ──
      case 'dailyHours':
        satisfied = dailyData.totalMinutes >= ach.condition.threshold * 60;
        break;

      case 'subjectDailyHours':
        satisfied = dailyData.subjectMinutes[ach.condition.subject!] >= ach.condition.threshold;
        break;

      // ── 晨行 / 夜航 ──
      case 'morningDays': {
        const count = allHistory.filter(d => d.firstActivityStartTime && d.firstActivityStartTime < '08:00').length;
        satisfied = count >= ach.condition.threshold;
        break;
      }
      case 'nightDays': {
        const count = allHistory.filter(d => d.lastActivityEndTime && d.lastActivityEndTime > '22:00').length;
        satisfied = count >= ach.condition.threshold;
        break;
      }

      // ── 极限专注 ──
      case 'focusSessions': {
        const daysWithLongSegment = allHistory.filter(d =>
          getLongestContinuousSegment(d) >= ach.condition.minMinutes
        ).length;
        satisfied = daysWithLongSegment >= ach.condition.threshold;
        break;
      }

      // ── 逆袭 ──
      case 'comeback': {
        const yesterday = allHistory[allHistory.length - 2]; // 倒数第二天
        if (!yesterday) break;
        const prevOk = yesterday.totalMinutes < ach.condition.prevMax;
        const todayOk = dailyData.totalMinutes >= ach.condition.todayMin;
        satisfied = prevOk && todayOk;
        break;
      }

      // ── 超额 ──
      case 'overachieveDays': {
        const subj = ach.condition.subject!;
        const goal = subjectGoals[subj] * 60; // 转为分钟
        const count = allHistory.filter(d => d.subjectMinutes[subj] >= goal * 1.5).length;
        satisfied = count >= ach.condition.threshold;
        break;
      }

      // ── 大满贯 ──
      case 'tripleOver':
        satisfied = ['physics', 'math', 'english'].every(subj => {
          const goal = subjectGoals[subj] * 60;
          return dailyData.subjectMinutes[subj] >= goal * 1.5;
        });
        break;

      // ── 三连绝世 ──
      case 'tripleStreak': {
        const last3 = allHistory.slice(-3);
        if (last3.length < 3) break;
        satisfied = last3.every(d =>
          ['physics', 'math', 'english'].every(subj => {
            const goal = subjectGoals[subj] * 60;
            return d.subjectMinutes[subj] >= goal * 1.5;
          })
        );
        break;
      }

      // ── 均衡日 ──
      case 'balanced': {
        const subs = ['physics', 'math', 'english'];
        satisfied = subs.every(subj => {
          const mins = dailyData.subjectMinutes[subj];
          const goal = subjectGoals[subj] * 60;
          return mins >= goal && mins < goal * 1.5;
        });
        break;
      }

      // ── 朝暮行 ──
      case 'dawnDusk':
        satisfied =
          dailyData.firstActivityStartTime !== null &&
          dailyData.firstActivityStartTime < '08:00' &&
          dailyData.lastActivityEndTime !== null &&
          dailyData.lastActivityEndTime > '22:00';
        break;
    }

    if (satisfied) {
      unlocked.push(ach);
      saveUnlocked(ach); // 写入 DB
    }
  }

  return unlocked;
}

// ============================================================
// 辅助函数（你需要根据实际数据结构实现）
// ============================================================

function sumTotalHours(history: DayData[]): number {
  return history.reduce((sum, d) => sum + d.totalMinutes, 0) / 60;
}

function sumSubjectHours(history: DayData[], subject: string): number {
  return history.reduce((sum, d) => sum + (d.subjectMinutes[subject] || 0), 0) / 60;
}

// 连续打卡天数：从今天往前数，直到遇到第一个 < 30min 的天
function getConsecutiveDays(history: DayData[]): number {
  let count = 0;
  const today = history[history.length - 1];
  // 今天还没学完 → 从昨天开始数（不断签）
  const startIdx = today.totalMinutes >= 30 ? history.length - 1 : history.length - 2;
  for (let i = startIdx; i >= 0; i--) {
    if (history[i].totalMinutes >= 30) count++;
    else break;
  }
  return count;
}

// 找当天最长的单段连续学习（相邻间隔 < 5min 算连续）
function getLongestContinuousSegment(day: DayData): number {
  if (!day.segments || day.segments.length === 0) return 0;
  let maxLen = day.segments[0].duration;
  let current = day.segments[0].duration;
  for (let i = 1; i < day.segments.length; i++) {
    const gap = day.segments[i].start - day.segments[i - 1].end;
    if (gap < 5) {
      current += day.segments[i].duration;
    } else {
      if (current > maxLen) maxLen = current;
      current = day.segments[i].duration;
    }
  }
  return Math.max(maxLen, current);
}
```

## 四、解锁 Toast

成就解锁时右下角弹出：

```
┌──────────────────────────┐
│  🌱  破土                 │
│  种子破土而出，            │
│  澜山之旅开始了 🌱        │
└──────────────────────────┘
```

- 背景 `#1e293b`，左边 4px 翠绿边框，圆角 8px
- 弹入：translateX(100%) → 0 + opacity 0→1，300ms ease-out
- 停留 3 秒
- 淡出：opacity 1→0，300ms ease-in
- 多个同时解锁 → 依次排队（上一个消失 0.3s 后弹出下一个）
